/**
 * Ingest daily P&L, normalize, and publish to BI layer.
 */
async function processDailyPnL(records) {
  // 1) Validate schema
  validate(records); // fragment 1
  // 2) Normalize currencies
  const normalized = await fxNormalize(records, 'USD'); // fragment 2
  // 3) Persist
  await saveToWarehouse(normalized); // fragment 3
  return { rows: normalized.length, status: 'ok' };
}

Return on Equity (ROE): \\( \\mathrm{ROE} = \\frac{\\text{Net\\ Income}}{\\text{Average\\ Shareholders'\\ Equity}} \\)

Discounted Cash Flow: \\( \\mathrm{PV} = \\sum_{t=1}^{T} \\frac{CF_t}{(1+r)^t} \\)

Use EPS for per-share profitability, ROE for capital efficiency, and DCF for intrinsic valuation context.
